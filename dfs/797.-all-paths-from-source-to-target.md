# 797. All Paths From Source to Target

```text
class Solution:
    def allPathsSourceTarget(self, graph: List[List[int]]) -> List[List[int]]:
        import queue
        q = queue.Queue()
        result = []
        self.degree = {}
        q.put((0, [0]))
        while q.qsize() != 0:
            node, path = q.get()
            if node == len(graph) - 1:
                result.append(path)
                continue
            for neighbor in graph[node]:
                q.put((neighbor, path + [neighbor]))
        return result
```
