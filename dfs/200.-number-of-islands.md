# 200. Number of Islands

straight forward

```text
class Solution:    def numIslands(self, grid: List[List[str]]) -> int:        self.visited = [[False for _ in range(len(grid[0]))] for _ in range(len(grid))]        moves = [[-1, 0], [1, 0], [0, 1], [0, -1]]        result = 0                for i in range(len(grid)):            for j in range(len(grid[0])):                                if not self.visited[i][j] and grid[i][j] == '1':                                 self.helper(grid, i, j, moves)                    result += 1                return result        def helper(self, grid, i, j, moves):        self.visited[i][j] = True        for move in moves:            moveX, moveY = move            newX, newY = i + moveX, j + moveY            if self.valid(newX, newY, self.visited) and grid[newX][newY] == '1':                self.helper(grid, newX, newY, moves)        return        def valid(self, i, j, visited):        return i >= 0 and j >= 0 and i < len(visited) and j < len(visited[0]) and visited[i][j] == False
```
