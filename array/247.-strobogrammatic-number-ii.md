# 247. Strobogrammatic Number II

注意先把中间加进去，再往两边加，这样少考虑很多corner case，还有就是两面不能为0

```text
class Solution(object):    def findStrobogrammatic(self, n):        """        :type n: int        :rtype: List[str]        """        result = []        if n == 1:            return ['0','1','8']        if n == 2:            return ['69','96','88','11']        cur = []        counter = 0        if n % 2 == 1:            cur = ['8','1','0']            counter = n - 1        if n % 2 == 0:            cur = ['69','96','88','11','00']            counter = n - 2                        while counter > 1:            tmp = []            for s in cur:                if counter == 3 or counter == 2:                    tmp.append('1'+s+'1')                    tmp.append('6'+s+'9')                    tmp.append('8'+s+'8')                    tmp.append('9'+s+'6')                else:                    tmp.append('0'+s+'0')                    tmp.append('1'+s+'1')                    tmp.append('6'+s+'9')                    tmp.append('8'+s+'8')                    tmp.append('9'+s+'6')            cur = tmp            counter -= 2          return cur
```
