# 60. Permutation Sequence

k = k - 1是关键

```text
class Solution:    def getPermutation(self, n: int, k: int) -> str:        factor = 1                #换成0based        k = k - 1        for i in range(1, n):            factor *= i                    used = [False] * (n)        ans = ''                for i in range(n):            idx = k // factor            k = k % factor            print(idx, k, ans, used, factor)            for j in range(n):                if not used[j]:                    if idx == 0:                        used[j] = True                        ans += str(j+1)                        break                    else:                        idx -= 1            if i < n - 1:                    factor = factor // (n - 1 - i);        return ans
```
