# 2. Add Two Numbers

注意在l1和l2中间一个为0退出循环的时候，carry要与不为0的list继续运算！可以用一些奇技淫巧，比如为空的话付值为0，继续运算，节约代码量！

```text
   while l1 or l2 or carry:
        v1 = v2 = 0        if l1:            v1 = l1.val            l1 = l1.next        if l2:            v2 = l2.val            l2 = l2.next
```

```text
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode(int x) { val = x; }
 * }
 */
 //
class Solution {
    public ListNode addTwoNumbers(ListNode l1, ListNode l2) {
        ListNode dummyNode = new ListNode(0);
        ListNode head = dummyNode;
        ListNode p1 = l1;
        ListNode p2 = l2;
        int carry = 0;
        //|| here not &&
        while (p1 != null || p2 != null || carry != 0) {
            
            int sum = ((p1!=null ? p1.val:0) + (p2!=null? p2.val:0) + carry);           
            ListNode tmp = new ListNode(sum%10);
            carry = sum /10;
            
            head.next = tmp;
            head = head.next;
            p1 =  p1 != null? p1.next : p1;
            p2 = p2 != null? p2.next : p2;
            
            
        }

        return dummyNode.next;
    }
}
```
