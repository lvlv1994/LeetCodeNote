# 269. Alien Dictionary

建立图，两两比较单词，第一个不相同的词建立边，然后topological sort



`class Solution(object):`

    `def alienOrder(self, words):`

        `"""`

        `:type words: List[str]`

        `:rtype: str`

        `"""`

        `adj = {}`

        `degree = {}`

        `for word in words:`

            `for char in word:`

                `degree[char] = 0`

      ``

        `for i in range(len(words)-1):`

            `cur = words[i]`

            `nexts = words[i+1]`

            `length = min(len(cur),len(nexts))`

            `for j in range(length):`

                `c1 = cur[j]`

                `c2 = nexts[j]`

                `if c1 != c2:`

                    `sets = set()`

                    `if c1 in adj:`

                        `sets = adj[c1]`

                    `if c2 not in sets:`

                        `sets.add(c2)`

                        `adj[c1] = sets`

                        `degree[c2] += 1`

                    `break`

        `import Queue`

        `q = Queue.Queue()`

        `for d in degree:`

            `if degree[d] == 0:`

                `q.put(d)`

        `result = ''`

        `while q.qsize()!=0:`

            `c = q.get()`

            `result += c`

            `if c in adj:`

                `for c2 in adj[c]:`

                    `degree[c2] -= 1`

                    `if degree[c2] == 0:`

                        `q.put(c2)`

        `if len(result) != len(degree):`

            `return ''`

        `return result`

            ``
