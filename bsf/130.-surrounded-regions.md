# 130. Surrounded Regions

注意在加入queue的时候就要标记，这样能避免这个点重复加入queue里

```text
class Solution:    def solve(self, board: List[List[str]]) -> None:        """        Do not return anything, modify board in-place instead.        """                if len(board) == 0:            return        self.directions = [[0, 1], [0, -1], [1, 0], [-1, 0]]        for i in range(len(board)):            self.bfs(board, i, 0)            self.bfs(board, i, len(board[0]) - 1)                for j in range(1, len(board[0])):            self.bfs(board, 0, j)            self.bfs(board, len(board) - 1, j)                    print(board)        for i in range(len(board)):            for j in range(len(board[0])):                if board[i][j] == 'O':                    board[i][j] = 'X'                if board[i][j] == 'F':                    board[i][j] = 'O'                        def bfs(self, board, i, j):        if board[i][j] != 'O':            return        import queue                q = queue.Queue()        q.put((i, j))        board[i][j] = 'F'        while q.qsize() != 0:            x, y = q.get()            for direction in self.directions:                newX, newY = x + direction[0], y + direction[1]                        if self.isValid(board, newX, newY):                //这个时候标记                    board[newX][newY] = 'F'                    q.put((newX, newY))                            def isValid(self, board, x, y):        return x >= 0 and y >= 0 and x < len(board) and y < len(board[0]) and board[x][y] == 'O'
```
