# 286. Walls and Gates

利用是否等于inf来判断是否visited过，注意因为是bfs 最先遍历到的一定是最近的门

```text
class Solution:    def wallsAndGates(self, rooms: List[List[int]]) -> None:        """        Do not return anything, modify rooms in-place instead.        """        import queue        import sys                q = queue.Queue()        moves = [[1, 0], [-1, 0], [0, 1], [0, -1]]                for i in range(len(rooms)):            for j in range(len(rooms[0])):                if rooms[i][j] == 0:                    q.put((i, j))                while q.qsize() != 0:            node = q.get()            x, y = node            for move in moves:                newX, newY = x + move[0], y + move[1]                if self.valid(newX, newY, rooms):                    rooms[newX][newY] = rooms[x][y] + 1                    q.put((newX, newY))    def valid(self, x, y, rooms):        return x >= 0 and y >= 0 and x < len(rooms) and y < len(rooms[0]) and rooms[x][y] == 2147483647
```
