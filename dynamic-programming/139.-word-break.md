# 139. Word Break

Just build a dp to record weather the word from i:i+j is in word dictionary

`class Solution(object):`

    `def wordBreak(self, s, wordDict):`

        `"""`

        `:type s: str`

        `:type wordDict: List[str]`

        `:rtype: bool`

        `"""`

        `dp = [False  for i in range(len(s)+1)]`

        `dp[0] = True`

        `for i in range(len(s)):`

            `for w in wordDict:`

                `if s[i-len(w)+1:].startswith(w) and dp[i-len(w)+1]:`

                    `dp[i+1] = True`

        `return dp[-1]`
