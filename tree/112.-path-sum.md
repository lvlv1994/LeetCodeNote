# 112. Path Sum

和257题思路差不多

```text
class Solution:    def hasPathSum(self, root: TreeNode, sum: int) -> bool:        self.result = False        if not root:            return self.result        self.helper(root, root.val, sum)        return self.result    def helper(self, root, sums, sum):        if root.left is None and root.right is None:            if sums == sum:                self.result = True            return        if root.left:            self.helper(root.left, sums + root.left.val, sum)        if root.right:            self.helper(root.right, sums + root.right.val, sum)
```

但是bfs的解法很有趣啊

```text
class Solution:    def hasPathSum(self, root, sum):        """        :type root: TreeNode        :type sum: int        :rtype: bool        """        if not root:            return False        de = [(root, sum - root.val), ]        while de:            node, curr_sum = de.pop()            if not node.left and not node.right and curr_sum == 0:                  return True            if node.right:                de.append((node.right, curr_sum - node.right.val))            if node.left:                de.append((node.left, curr_sum - node.left.val))        return False
```
