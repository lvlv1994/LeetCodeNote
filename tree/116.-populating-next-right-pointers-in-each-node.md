# 116. Populating Next Right Pointers in Each Node

```text
class Solution:
    def connect(self, root: 'Node') -> 'Node':
        import queue
        if root is None:
            return None
        q = queue.Queue()
        q.put(root)
        head = root
        while q.qsize() != 0:
            size = q.qsize()
            prev = q.get()
            if prev.left is not None:
                q.put(prev.left)
            if prev.right is not None:
                q.put(prev.right)
            for i in range(size-1):
                node = q.get()
                if node.left is not None:
                    q.put(node.left)
                if node.right is not None:
                    q.put(node.right)
                prev.next = node
                prev = node
        return head
```
