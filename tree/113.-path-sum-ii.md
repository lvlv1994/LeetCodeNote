```text
class Solution:    def pathSum(self, root: TreeNode, sum: int) -> List[List[int]]:        self.result = []        if not root:            return []        self.helper(root, root.val, sum, [root.val])        return self.result    def helper(self, root, sums, sum, path):        if root.left is None and root.right is None:            if sums == sum:                self.result.append(path[:])        if root.left:            path.append(root.left.val)            self.helper(root.left, sums + root.left.val, sum, path)            path.pop()        if root.right:            path.append(root.right.val)            self.helper(root.right, sums + root.right.val, sum, path)            path.pop()
```
