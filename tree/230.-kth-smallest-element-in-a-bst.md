`class Solution:`

    `def kthSmallest(self, root, k):`

        `"""`

        `:type root: TreeNode`

        `:type k: int`

        `:rtype: int`

        `"""`

        `if root is None:`

            `return` 

        `self.ans = 0`

        `self.counter = 0`

        `self.helper(root,k)`

        `return self.ans`

        ``

    `def helper(self,root,k):`

        `if root is None:`

            `return` 

        `print(root.val)`

        ``

        `self.helper(root.left,k)`

        `self.counter += 1`  

        `print(root.val,'after')`

        `if self.counter == k:`

            `self.ans = root.val`

            `return`

        `self.helper(root.right,k)`
