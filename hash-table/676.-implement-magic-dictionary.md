# 676. Implement Magic Dictionary

用一个字典记录所有单词少一个字母为key，少字母的为止及字母value，然后check

```text
class MagicDictionary(object):    def __init__(self):        """        Initialize your data structure here.        """        self.magicdic = {}            def buildDict(self, dict):        """        Build a dictionary through a list of words        :type dict: List[str]        :rtype: void        """            for d in dict:            for i in range(len(d)):                key = d[:i]+d[i+1:]                if key not in self.magicdic:                    self.magicdic[key] = []                    self.magicdic[key].append([i,d[i]])                else:                     self.magicdic[key].append([i,d[i]])                        #print(self.magicdic)    def search(self, word):        """        Returns if there is any word in the trie that equals to the given word after modifying exactly one character        :type word: str        :rtype: bool        """        #print(word)        for i in range(len(word)):            new_word = word[:i] + word[i+1:]            if new_word in self.magicdic:                for n in self.magicdic[new_word]:                    #print(n,i)                    if i == n[0] and n[1] != word[i]:                                            return True        return False
```
