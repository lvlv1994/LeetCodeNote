# 282. Expression Add Operators

两个知识点

1. 处理乘法时利用lastFactor，先减后加，这样比记录mulitplier更简单
2. \`if len\(num\[pos: i+1\]\) &gt; 1 and num\[pos: i+1\]\[0\] == '0': continue \#case 1 \* 05\` 这个case的判断

```text
class Solution:    def addOperators(self, num: str, target: int) -> List[str]:        self.result = []        self.helper(num, 0, 0, '', target, 0)        return self.result        def helper(self, num, result, pos, ans, target, lastF):        if pos == len(num):            if result == target:                self.result.append(ans)                return        for i in range(pos, len(num)):            subStr = int(num[pos: i+1])            if len(num[pos: i+1]) > 1 and num[pos: i+1][0] == '0': continue #case 1 * 05            if pos == 0:                self.helper(num, subStr , i + 1, str(subStr), target, subStr)            else:                self.helper(num, subStr + result , i + 1, ans + '+' + str(subStr), target, subStr)                self.helper(num, -subStr + result , i + 1, ans + '-' + str(subStr), target, -subStr)                self.helper(num, result - lastF + lastF * subStr , i + 1, ans + '*' + str(subStr), target, lastF * subStr)        return    
```
