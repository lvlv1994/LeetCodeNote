```text
class Solution:    def combinationSum2(self, candidates: List[int], target: int) -> List[List[int]]:        self.result = []        candidates = sorted(candidates)        self.helper(candidates, target, [], 0)        return self.result        def helper(self, candidates, target, ans, pos):           if target < 0:            return                if target == 0:            self.result.append(ans[:])                for i in range(pos, len(candidates)):            if candidates[i] > target:                break            if i != pos and candidates[i] == candidates[i-1]:                continue            ans.append(candidates[i])            self.helper(candidates, target - candidates[i], ans, i+1)            ans.pop()                      return
```
