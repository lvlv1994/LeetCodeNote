# 39. Combination Sum

`class Solution(object):`

    `def combinationSum(self, candidates, target):`

        `"""`

        `:type candidates: List[int]`

        `:type target: int`

        `:rtype: List[List[int]]`

        `"""`

        `self.result = []`

        `self.ans = []`

        `candidates = sorted(candidates)`

        `self.dfs(target,candidates,0)`

        `return self.result`

    `def dfs(self,target,candidates,pos):`

        `if target == 0:`        

            `self.result.append(self.ans[:])`       

            `return`

        `for i in range(pos,len(candidates)):`

            `if target - i >= 0:`

                `self.ans.append(candidates[i])`

                `self.dfs(target-candidates[i],candidates,i)`

                `self.ans.pop()`
