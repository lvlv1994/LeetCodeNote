# 351. Android Unlock Patterns

Backtracking problem, 1357 and 2468 are symmetric, we should build a skip dic.

```text
class Solution:    def numberOfPatterns(self, m: int, n: int) -> int:        skip = [[0 for j in range(10)] for i in range(10)]        skip[1][3] = skip[3][1] = 2        skip[1][7] = skip[7][1] = 4        skip[3][9] = skip[9][3] = 6        skip[7][9] = skip[9][7] = 8        skip[1][9] = skip[9][1] = skip[2][8] = skip[8][2] = skip[3][7] = skip[7][3] = skip[4][6] = skip[6][4] = 5        vis = [False for i in range(10)]                result = 0        #找m 到 n个键        for i in range(m, n + 1):            #1，3，7，9 是一样的            result += self.helper(1, i - 1, skip, vis) * 4            #2，4，6，8 是一样的            result += self.helper(4, i - 1, skip, vis) * 4            result += self.helper(5, i - 1, skip, vis)        return result        def helper(self, start, remain, skip, vis):        if remain < 0:            return 0        if remain == 0:            return 1        vis[start] = True                result = 0        for i in range(1, 10):            #这个判断条件很重要            if not vis[i] and (skip[start][i] == 0 or (vis[skip[start][i]])):                result += self.helper(i, remain - 1, skip, vis)        vis[start] = False        return result
```
